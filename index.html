    /** Inserir texto dentro de um elemento com o ID "meuElemento" **/
    let elemento = document.getElementById("meuElemento");
    let range = document.createRange();

    /** Seleciona todo o conteúdo do elemento **/
    range.selectNodeContents(elemento); 
    let novoTexto = document.createTextNode("Novo texto inserido");

    /** Remove o conteúdo selecionado **/
    range.deleteContents(); 

    /** Insere o novo texto no lugar do conteúdo removido **/
    range.insertNode(novoTexto);
